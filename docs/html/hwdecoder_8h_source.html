41 static enum AVPixelFormat find_fmt_by_hw_type(const enum AVHWDeviceType type)
42 {
43  enum AVPixelFormat fmt;
44  switch (type) {
45  case AV_HWDEVICE_TYPE_VAAPI:
46  fmt = AV_PIX_FMT_VAAPI;
47  break;
48  case AV_HWDEVICE_TYPE_DXVA2:
49  fmt = AV_PIX_FMT_DXVA2_VLD;
50  break;
51  case AV_HWDEVICE_TYPE_D3D11VA:
52  fmt = AV_PIX_FMT_D3D11;
53  break;
54  case AV_HWDEVICE_TYPE_VDPAU:
55  fmt = AV_PIX_FMT_VDPAU;
56  break;
57  case AV_HWDEVICE_TYPE_VIDEOTOOLBOX:
58  fmt = AV_PIX_FMT_VIDEOTOOLBOX;
59  break;
60  default:
61  fmt = AV_PIX_FMT_NONE;
62  break;
63  }
64  return fmt;
65 }

67 static enum AVPixelFormat get_vaapi_format(AVCodecContext *ctx,
68  const enum AVPixelFormat *pix_fmts)
69 {
70  const enum AVPixelFormat *p;
71  for (p = pix_fmts; *p != AV_PIX_FMT_NONE; p++) {
72  if (*p == AV_PIX_FMT_VAAPI)
73  return *p;
74  }
75  avthreadlogger.log(LogLevel::fatal) << "get_vaapi_format: Unable to decode this stream using VA-API" << std::endl;
76  return AV_PIX_FMT_NONE;
77 }
